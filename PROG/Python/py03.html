from math import sqrt

def positive_root(a, b, c):
    return (-b + sqrt(b**2 - 4*a*c)) / (2*a)

# Math/03_quad.py
from math import sqrt

def quad(a,b,c):
    D = b**2 - 4*a*c
    return ((-b + sqrt(D)) / (2*a), (-b - sqrt(D)) / (2*a))
  
print(quad(1,2,-3))
print(quad(1,2,1))
print(quad(1,2,3))

Discriminant as Class

# Math/06_discr_class.py
class Discriminator:
    # Initializer
    def __init__(self, a, b, c):
        self.D = b**2 - 4*a*c
    
    def is_real(self):
        return self.D > 0
    
    def is_repeated(self):
        return self.D == 0
      
    def is_complex(self):
        return self.D < 0
      
d = Discriminator(1, 2, -1)

print(d.is_real())
print(d.is_repeated())
print(d.is_complex())

Linear Equation

# Math/07_linear.py
class Linear:
    def __init__(self, a, b):
        self.a = a
        self.b = b
    
    def slope(self):
        return self.a
    
    def x_intercept(self):
        return -self.b / self.a
      
    def y_intercept(self):
        return self.b
      
l = Linear(2, 1)

print(l.slope())
print(l.x_intercept())
print(l.y_intercept())
